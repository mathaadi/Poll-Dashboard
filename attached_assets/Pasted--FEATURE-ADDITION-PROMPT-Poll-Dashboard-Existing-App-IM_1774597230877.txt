## FEATURE 3 — Multilingual Feedback Translation Before NLP

**Where:** `src/services/nlp.js` and the upload route `src/routes/upload.js`

**What to do:**
Before running sentiment classification and theme extraction on feedback text, detect if the text is non-English and translate it to English using the **Anthropic Claude API** (you already have API access in this environment).

### Step 1: Add translation function in `nlp.js`

```javascript
async function translateToEnglish(text) {
  // Skip if text is empty or clearly English
  if (!text || text.trim().length === 0) return text;
  
  // Quick heuristic: if text only contains ASCII + common punctuation, assume English
  const nonAsciiRatio = (text.match(/[^\x00-\x7F]/g) || []).length / text.length;
  
  // If mostly ASCII AND no Devanagari/Arabic/Chinese/other script chars → skip translation
  const hasNonLatinScript = /[\u0900-\u097F\u0600-\u06FF\u4E00-\u9FFF\u0400-\u04FF\uAC00-\uD7AF]/.test(text);
  
  if (nonAsciiRatio < 0.15 && !hasNonLatinScript) return text; // Already English
  
  try {
    const response = await fetch("https://api.anthropic.com/v1/messages", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        model: "claude-sonnet-4-20250514",
        max_tokens: 200,
        messages: [{
          role: "user",
          content: `Translate the following student feedback to English. Return ONLY the translated text, nothing else. If it is already in English, return it unchanged.\n\nFeedback: "${text}"`
        }]
      })
    });
    const data = await response.json();
    const translated = data.content?.[0]?.text?.trim();
    return translated || text; // fallback to original if API fails
  } catch (err) {
    console.error('Translation error:', err.message);
    return text; // fallback to original
  }
}
```

### Step 2: Update the main classify function to be async

```javascript
// Change classifyFeedback to async:
async function classifyFeedback(originalText) {
  if (!isUsefulFeedback(originalText)) {
    return { is_useful: false, sentiment: null, themes: [], translated_text: null };

  // Translate if needed
  const text = await translateToEnglish(originalText);

  return {

    sentiment,
    themes: themes.join(','),
    translated_text: text !== originalText ? text : null // only store if actually translated
  };
}
```
